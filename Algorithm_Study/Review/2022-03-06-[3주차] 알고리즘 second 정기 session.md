class Solution(object):
    def numIslands(self, grid):
        # λ°ν™ν•  κ°’
        result = 0
        
        # λ¨λ“  matrixλ¥Ό λλ©΄μ„ μƒλ΅μ΄ island μ—¬λ¶€ ν™•μΈ 
        for i in range(len(grid)):
            for j in range(len(grid[0])):
                # ν•΄λ‹Ή κ°’μ΄ 1λ΅, μƒλ΅μ΄ island κ°€λ¥ν•λ‹¤λ©΄ dfs_recursion μ‹μ‘
                if grid[i][j] == "1":
                    self.dfs_recursion(grid, i, j)
                    # λ¨λ“  recursion μΆ…λ£λλ©΄ ν•λ‚μ island μ°Ύμ€ κ²ƒμ΄λ―€λ΅ resultμ— 1 μ¶”κ°€ν•΄ μ¤€λ‹¤.
                    result += 1
        
        return result
                                    
    # dfs recursion μ§„ν–‰ν•  μ¶”κ°€ ν•¨μ    
    def dfs_recursion(self, matrix, current_i, current_j):
        # edge case : recursionμ„ returnν•  κµ¬μ‹¤μ΄ λλ” μ΅°κ±΄λ“¤
        # matrix λ²”μ„ λ°–μ΄κ±°λ‚, ν•΄λ‹Ή κ°’μ΄ 0μΌ λ•
        if (current_i < 0 or current_i >= len(matrix) or current_j < 0 or current_j >= len(matrix[0]) or matrix[current_i][current_j] == "0"):
            return
        
        # markν•κΈ° μ„ν•΄ ν•΄λ‹Ή κ°’μ€ 0μΌλ΅ κµμ²΄
        matrix[current_i][current_j] = "0"
        
        direction = [[-1, 0], [0, 1], [1, 0], [0, -1]]
        
        # μλ” κ³³ κΈ°μ¤€μΌλ΅ μ„μ•„λ, μ¤λ¥Έμ½, μ™Όμ½ λ¨λ‘ recursion μ§„ν–‰
        for i in range(len(direction)):
            self.dfs_recursion(matrix, current_i - direction[i][0], current_j - direction[i][1])
